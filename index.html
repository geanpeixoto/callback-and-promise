Callbacks

Callback é a referência à uma função que será executada no futuro, em um tempo conveniênte. 

						
/**
 * Irá executar um método qualquer após o delay de 2s
 *
 * @param {function} callback
 * 		Método que será executado
 *
 * @return {voidnoc}
 */
function async(callback) {
	setTimeout(function() {
		callback(Math.trunc(Math.random()*1000));
	}, 2000);
}
